[USERテーブル作成]
CREATE TABLE  USER(
  USER_ID VARCHAR(12) PRIMARY KEY,
  PASSWORD VARCHAR(20),
  NAME VARCHAR(20),
  HEIGHT DOUBLE ,
  WEIGHT DOUBLE ,
  TARGET_WEIGHT DOUBLE 
);

[MANAGEテーブル作成]
CREATE TABLE MANAGE (
  USER_ID VARCHAR(12) PRIMARY KEY,
  DATE VARCHAR,
  BREAKFAST VARCHAR(5),
  BFTEXT VARCHAR(100),
  LUNCH VARCHAR(5),
  LCTEXT VARCHAR(100),
  DINNER VARCHAR(5),
  DNTEXT VARCHAR(100),
  SNACK INT(1),
  EXERCISE INT(1),
  DRINK INT(1),
  DAYWEIGHT DOUBLE,
  PICTURE VARCHAR,
  BMI DOUBLE,
  COUNTER VARCHAR DEFAULT 0
); 

[POINTテーブル作成]
CREATE TABLE POINT (
  USER_ID VARCHAR(12) PRIMARY KEY,
  CPOINT INT,
  PPOINT INT
);

[BBSテーブル作成]
CREATE TABLE BBS (
